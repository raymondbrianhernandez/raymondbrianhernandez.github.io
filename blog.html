Zoom Buttons

February 15, 2021

View GitHub Repository

    Build of Materials (BOM):

  • ATTiny85 Microcontroller
  • USB cable
  • (3x) Momentary Switch
  • (3x) 10k Resistors

                                            
#include OneButton.h
#include DigiKeyboardFr.h

int button1pin = 0;
int button2pin = 3;
int button3pin = 2;
OneButton button1(button1pin, true);
OneButton button2(button2pin, true);
OneButton button3(button3pin, true);

void setup() {
  button1.attachClick(click1);
  button2.attachClick(click2);
  button3.attachClick(click3);

  DigiKeyboard.sendKeyStroke(0);
  DigiKeyboard.delay(500);
}

void loop() {
  button1.tick();
  button2.tick();
  button3.tick();
}

// Alt-A - Turn On/Off Microphone
void click1() {
  DigiKeyboard.sendKeyStroke(0);
  DigiKeyboard.sendKeyStroke(0, MOD_SHIFT_LEFT | MOD_CONTROL_LEFT | MOD_ALT_LEFT);
  DigiKeyboard.delay(100); 
  DigiKeyboard.sendKeyStroke(KEY_A, MOD_ALT_LEFT);
} 

// Alt-Y - Raise/Lower Hand
void click2() {
  DigiKeyboard.sendKeyStroke(0);
  DigiKeyboard.sendKeyStroke(0, MOD_SHIFT_LEFT | MOD_CONTROL_LEFT | MOD_ALT_LEFT);
  DigiKeyboard.delay(100); 
  DigiKeyboard.sendKeyStroke(KEY_Y, MOD_ALT_LEFT);
} 

// Alt-V - Turn On/Off Video
void click3() {
  DigiKeyboard.sendKeyStroke(0);
  DigiKeyboard.sendKeyStroke(0, MOD_SHIFT_LEFT | MOD_CONTROL_LEFT | MOD_ALT_LEFT);
  DigiKeyboard.delay(100); 
  DigiKeyboard.sendKeyStroke(KEY_V, MOD_ALT_LEFT);
}

The ATTiny85 board is compatible with the Arduino Bootloader to flash the firmware. The schematics is straightforward. Three 10k pull-up resistors are needed to avoid floating points. A mometary "tap" to ground by the momentary swicth will call the click() function to invoke the built-in Zoom shorthand commands. Alt+A for audio, Alt+V for Video, and Alt+Y to raise hands.
